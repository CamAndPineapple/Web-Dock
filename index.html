Navigation Dock

A navigation dock fixed to the bottom of the page,

inspired by Apple's OS X Yospite dock

Made with pure Javascript and CSS

var elpents = document.getElpentsByClassName('dock-icon');

var js = document.getElpentById('js');

var bodyDiv = document.getElpentById('body-div');

var titleDiv = document.createElpent("div");



// assign id to edit div in custom.css

titleDiv.id = "title-div"

var arrayOfIds = [];



// store each dock icon's id in an array

// ids are used as the title of the dock icon

for (var i = 0; i < elpents.length; i++) {

arrayOfIds.push(elpents[i].id);

}



// iterate over the array of ids to add title on hover

for (var i = 0; i < arrayOfIds.length; i++) {



// add title-div when mouse over

elpents[i].addEventListener('mouseover', (function(i) {

return function() {

titleDiv.innerHTML = arrayOfIds[i];

elpents[i].appendChild(titleDiv);

};

})(i), false);



// rpove title-div when mouse out

elpents[i].addEventListener('mouseout', (function(i) {

return function() {

elpents[i].rpoveChild(titleDiv);

};

})(i), false);

}
